four_yearly_country_pulse = {
	on_actions = { fu_clear_situations }
}

on_game_start = {
	on_actions = {
		delay = { days =  1 }
		fu_game_setup
	}
}

fu_game_setup = {
	effect = {
		set_global_variable = is_fu_game
		if = {
			limit = {
				has_game_rule = fu_remove_sop_countries_setting
			}
			fu_remove_sop_effect = yes
		}
		if = {
			limit = {
				has_game_rule = fu_europe_europe_centric_setting
			}
			fu_europe_centric_effect = yes
		}
		if = {
			limit = {
				has_game_rule = fu_asia_centric_setting
			}
			fu_asia_centric_effect = yes
		}
		if = {
			limit = {
				has_game_rule = fu_remove_low_importance_setting
			}
			fu_remove_low_importance_nations_effect = yes
		}
		if = {
			limit = {
				has_game_rule = fu_remove_unowned_pops_setting
			}
			fu_remove_unowned_pops = yes
		}
	}
}

fu_clear_situations = {
	trigger = {
		is_ai = no
		is_situation_active = situation:great_pestilence
		or = {
			has_game_rule = fu_europe_europe_centric_setting
			has_game_rule = fu_asia_centric_setting
		}
	}
	effect = {
		end_situation = situation:great_pestilence
	}
}
