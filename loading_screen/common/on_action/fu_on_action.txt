monthly_country_pulse = {
	on_actions = { fu_on_game_start }
}

fu_on_game_start = {
	trigger = {
		NOT = {
			has_global_variable = fu_setup_complete
		}
	}
	effect = {
		set_global_variable = is_fu_game
		if = {
			limit = {
				has_game_rule = fu_cull_sop_countries_setting
			}
			fu_cull_sop_effect = yes
		}
		if = {
			limit = {
				has_game_rule = fu_europe_europe_centric_setting
			}
			fu_europe_centric_effect = yes
		}
		if = {
			limit = {
				has_game_rule = fu_asia_centric_setting
			}
			fu_asia_centric_effect = yes
		}
		if = {
			limit = {
				has_game_rule = fu_remove_low_importance_setting
			}
			fu_remove_low_importance_nations_effect = yes
		}
		set_global_variable = fu_setup_complete
	}
}
