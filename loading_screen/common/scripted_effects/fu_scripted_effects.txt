fu_cull_sop_effect = {
	every_location_in_the_world = {
		limit = {
			has_owner = no
		}
		every_pop = {
			destroy_pop = this
		}
	}
}

fu_europe_centric_effect = {
	if = {
		limit = { has_game_rule = fu_recolonization_blocked_setting }
		sub_continent:east_asia = { fu_remove_and_disable_sub_continent = yes }
		sub_continent:south_asia = { fu_remove_and_disable_sub_continent = yes }
		sub_continent:south_east_asia = { fu_remove_and_disable_sub_continent = yes }
		sub_continent:australasia = { fu_remove_and_disable_sub_continent = yes }
		sub_continent:pacific_islands = { fu_remove_and_disable_sub_continent = yes }
	} else = {
		sub_continent:east_asia = { fu_remove_sub_continent = yes }
		sub_continent:south_asia = { fu_remove_sub_continent = yes }
		sub_continent:south_east_asia = { fu_remove_sub_continent = yes }
		sub_continent:australasia = { fu_remove_sub_continent = yes }
		sub_continent:pacific_islands = { fu_remove_sub_continent = yes }
	}
	sub_continent:south_america = { fu_remove_sub_continent = yes }
}

fu_asia_centric_effect = {
	if = {
		limit = { has_game_rule = fu_recolonization_blocked_setting }
		sub_continent:western_europe = { fu_remove_and_disable_sub_continent = yes }
		sub_continent:eastern_europe = { fu_remove_and_disable_sub_continent = yes }
		sub_continent:north_africa = { fu_remove_and_disable_sub_continent = yes }
		sub_continent:west_africa = { fu_remove_and_disable_sub_continent = yes }
		sub_continent:central_africa = { fu_remove_and_disable_sub_continent = yes }
	}
	else = {
		sub_continent:western_europe = { fu_remove_sub_continent = yes }
		sub_continent:eastern_europe = { fu_remove_sub_continent = yes }
		sub_continent:north_africa = { fu_remove_sub_continent = yes }
		sub_continent:west_africa = { fu_remove_sub_continent = yes }
		sub_continent:central_africa = { fu_remove_sub_continent = yes }
	}
	sub_continent:north_america = { fu_remove_sub_continent = yes }
	sub_continent:south_america = { fu_remove_sub_continent = yes }
}

fu_remove_low_importance_nations_effect = {
	sub_continent:pacific_islands = { fu_remove_sub_continent = yes }
	sub_continent:north_america = { fu_remove_sub_continent = yes }
	sub_continent:south_america = { fu_remove_sub_continent = yes }
}

fu_remove_sub_continent = {
	every_location_in_sub_continent = {
		if = {
			limit = {
				fu_can_remove_location = yes
			}
			fu_remove_location = yes
		}
	}
}

fu_remove_and_disable_sub_continent = {
	every_location_in_sub_continent = {
		add_location_modifier = {
			modifier = fu_location_disabled_static_modifier
			years = -1
			mode = add
		}
		fu_remove_location = yes
	}
}

fu_remove_location = {
	every_buildings_in_location = {
		change_building_level = -10
	}
	every_pop = {
		destroy_pop = this
	}
	if = {
		limit = {
			has_owner = yes
		}
		owner = {
			abandon_location = prev
		}
	}
}
