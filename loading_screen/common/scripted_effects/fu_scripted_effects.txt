fu_cull_sop_effect = {
	every_location_in_the_world = {
		limit = {
			has_owner = no
		}
		every_pop = {
			destroy_pop = this
		}
	}
}

	sub_continent:east_asia = { fu_remove_sub_continent = yes }

fu_remove_low_importance_nations_effect = {
	sub_continent:pacific_islands = { fu_remove_sub_continent = yes }
	sub_continent:north_america = { fu_remove_sub_continent = yes }
	sub_continent:south_america = { fu_remove_sub_continent = yes }
}

fu_remove_sub_continent = {
	every_location_in_sub_continent = {
		every_buildings_in_location = {
			change_building_level = -10
		}
		every_pop = {
			destroy_pop = this
		}
		if = {
			limit = {
				has_owner = yes
			}
			owner = {
				abandon_location = prev
			}
		}
	}
}
