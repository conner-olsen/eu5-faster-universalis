### Combat Speed ###
# Note to those who don't understand where these values come from:
# In vanilla, the ticks are bi-hourly
# Thus there are 12 bi-hourly ticks a day.
# I am using 47 hour ticks, which would be 23.5x less ticks.
fu_vanilla_combat_speed_adjustment = { # Needs to be adjusted by ((47/2)/(50/11))=5.17
	game_data = {
		category = country
	}
	naval_damage_done = 5.17
	army_infantry_power = 5.17
	army_cavalry_power = 5.17
	army_artillery_power = 5.17
	army_auxiliary_power = 5.17
	combat_speed_modifier = 5.17
	army_initiative = 5.17
	navy_initiative = 5.17
}
